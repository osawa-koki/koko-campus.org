関数を値として扱う講座第二弾

機能は関数の戻り値として関数を返すところまで説明しました。

コードをもう一度提示します。 function add(x, y) { return x + y; } function times(x, y) { return x * y; } function add_times(fx1, fx2) { return function (a, b, c) { return fx2(fx1(a, b), c) } } console.log(add_times(add, times)(2, 3, 4)); console.log(add_times(times, add)(2, 3, 4)); このうちの、「add_times(..., ...)」の部分は引数として受け取った関数2つを合成して返しています。
これが関数の合成です。

しかしながら、毎回これを書くのは大変です。
ということで、これを別の変数に代入しましょう♪ function add(x, y) { return x + y; } function times(x, y) { return x * y; } function add_times(fx1, fx2) { return function (a, b, c) { return fx2(fx1(a, b), c) } } const add_times1 = add_times(add, times); // 関数の合成 const add_times2 = add_times(times, add); // 関数の合成 console.log(add_times1(2, 3, 4)); // 20 console.log(add_times2(2, 3, 4)); // 10 合成した関数を別の関数として扱うことも出来るようになりました。
