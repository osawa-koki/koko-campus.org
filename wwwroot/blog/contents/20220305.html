isset関数では深い配列に対しても有効です。 $jp = array( "kanto" => array( "saitama" => [ "soka", "koshigaya", "kawaguchi" ], "tokyo" => [ "adachi", "chiyoda", "shibuya" ] ) ); var_dump(isset($jp)); var_dump(isset($jp["kanto"])); var_dump(isset($jp["kanto"]["saitama"])); var_dump(isset($jp["kanto"]["saitama"]["soka"])); var_dump(isset($jp["kanto"]["saitama"]["soka"]["teshiro"])); /* ↓ コンソール ↓ bool(true) bool(true) bool(true) bool(false) bool(false) */
